\begin{align*}
  v^\mu &= \frac{\textup{d}x^\mu}{\textup{d}\tau}\\
        &= \gamma \frac{\textup{d}}{\textup{d}t}(\textup{c}t, x, y, z)\\
        &= \gamma \left ( \textup{c}\frac{\textup{d}t}{\textup{d}t}, \frac{\textup{d}x}{\textup{d}t}, \frac{\textup{d}y}{\textup{d}t}, \frac{\textup{d}z}{\textup{d}t} \right )
         = \gamma \left (\textup{c}, \frac{\textup{d}x}{\textup{d}t},
                          \frac{\textup{d}y}{\textup{d}t},
                          \frac{\textup{d}z}{\textup{d}t}  \right )
         = \gamma (\textup{c}, \dot{x}, \dot{y}, \dot{z})\\
        &= \gamma (\textup{c}, \textbf{v})
         = \gamma \begin{pmatrix} \textup{c} \\ \textbf{v}\end{pmatrix}
         = \begin{pmatrix} \gamma \textup{c} \\ \gamma \textbf{v}\end{pmatrix}
\end{align}
